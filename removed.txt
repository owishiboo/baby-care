void change(JFXButton button){
        button.setOnMouseEntered(e -> button.setStyle("-fx-text-fill: #ffffff;-fx-background-color: FFCED0;"));
        button.setOnMouseExited(e -> button.setStyle("-fx-text-fill: #42070b; -fx-background-color: FFCED0;"));
}
